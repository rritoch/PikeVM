Protocols.HTTP.Query()->`()()

Class Protocols.HTTP.Query

Description

Open and execute an HTTP query.

Example
HTTP.Query o=HTTP.Query();


 void ok()
 {
    write("ok...\n");
    write("%O\n", o->headers);
    exit(0);
 }


 void fail()
 {
    write("fail\n");
    exit(0);
 }


 int main()
 {
    o->set_callbacks(ok, fail);
    o->async_request("pike.ida.liu.se", 80, "HEAD / HTTP/1.0");
    return -1;
 }
